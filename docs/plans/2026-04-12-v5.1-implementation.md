    window._heatLayer = heat;
    
    // Salon markers
    for (const [pos, coords] of Object.entries(D.salon_gps)) {

   function makeDrillableChart(canvasId, monthlyData, config) {
       chartDrillState[canvasId] = { level: 'monthly', data: monthlyData };
